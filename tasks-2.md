```python
def is_isomorphic(s: str, t: str) -> bool:
    if len(s) != len(t):
        return False
    
    # –°–ª–æ–≤–∞—Ä—å: –∫–∞–∫–æ–π —Å–∏–º–≤–æ–ª –∏–∑ s —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É–µ—Ç —Å–∏–º–≤–æ–ª—É –≤ t
    s_to_t = {}
    # –°–ª–æ–≤–∞—Ä—å: –∫–∞–∫–æ–π —Å–∏–º–≤–æ–ª –∏–∑ t –ø—Ä–∏—à—ë–ª –æ—Ç —Å–∏–º–≤–æ–ª–∞ –≤ s
    t_to_s = {}
    
    for a, b in zip(s, t):
        # –ï—Å–ª–∏ —Å–∏–º–≤–æ–ª –∏–∑ s —É–∂–µ –≤—Å—Ç—Ä–µ—á–∞–ª—Å—è
        if a in s_to_t:
            if s_to_t[a] != b:
                return False  # –ù–æ –æ–Ω –¥–æ–ª–∂–µ–Ω –±—ã–ª —Å—Ç–∞—Ç—å –¥—Ä—É–≥–∏–º ‚Äî –æ—à–∏–±–∫–∞!
        # –ï—Å–ª–∏ —Å–∏–º–≤–æ–ª –∏–∑ t —É–∂–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–ª—Å—è, –Ω–æ –Ω–µ –¥–ª—è a
        elif b in t_to_s:
            if t_to_s[b] != a:
                return False  # –≠—Ç–æ—Ç —Å–∏–º–≤–æ–ª –≤ t —É–∂–µ "–∑–∞–Ω—è—Ç" –¥—Ä—É–≥–∏–º
        # –û–±–∞ —Å–∏–º–≤–æ–ª–∞ –Ω–æ–≤—ã–µ ‚Äî —Ñ–∏–∫—Å–∏—Ä—É–µ–º —Å–≤—è–∑—å
        else:
            s_to_t[a] = b
            t_to_s[b] = a

    return True
```

### –†–µ–∞–ª–∏–∑–∞—Ü–∏—è

```python
def missing_number(nums: list) -> int | None:
    if not nums:
        return None

    size = len(nums)
    base = nums[0]
    left = 0
    right = size - 1

    # –ï—Å–ª–∏ –ø–æ—Å–ª–µ–¥–Ω–∏–π —ç–ª–µ–º–µ–Ω—Ç "–Ω–∞ —Å–≤–æ—ë–º –º–µ—Å—Ç–µ", –∑–Ω–∞—á–∏—Ç, –ø—Ä–æ–ø—É—Å–∫–∞ –≤–Ω—É—Ç—Ä–∏ –Ω–µ—Ç
    if nums[right] == base + right:
        return None

    # –ë–∏–Ω–∞—Ä–Ω—ã–π –ø–æ–∏—Å–∫ –ø–µ—Ä–≤–æ–≥–æ –º–µ—Å—Ç–∞, –≥–¥–µ –Ω–∞—Ä—É—à–µ–Ω –ø–æ—Ä—è–¥–æ–∫
    while left < right:
        # –ù–∞—Ö–æ–¥–∏–º —Å–µ—Ä–µ–¥–∏–Ω—É —Å–ø–∏—Å–∫–∞
        index = (left + right) // 2

        # –ï—Å–ª–∏ –∑–Ω–∞—á–µ–Ω–∏–µ –±–æ–ª—å—à–µ –æ–∂–∏–¥–∞–µ–º–æ–≥–æ, –∑–Ω–∞—á–∏—Ç, –ø—Ä–æ–ø—É—Å–∫ –≤ –ª–µ–≤–æ–π –ø–æ–ª–æ–≤–∏–Ω–µ
        if nums[index] - base > index:
            right = index
        else:
            left = index + 1
            
    return base + left
```
